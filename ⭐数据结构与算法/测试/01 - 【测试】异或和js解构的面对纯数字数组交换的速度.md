## 对象
- 异或`^`交换
- js解构`[ ] = [ ]`
## 条件
- 利用web api中的`performance.now()`测试
- 测试次数1000次，数组长度1到100000随机，值从0到1000

## 数据

|     | 异或交换    | 解构交换    |
| --- | ------- | ------- |
| 时间  | 0.085ms | 0.256ms |


## 结论： 异或交换的速度是异或交换速度的接近三倍。
## 测试代码
```js
    function swapXor(arr, a, b) {
      arr[a] = arr[a] ^ arr[b];
      arr[b] = arr[a] ^ arr[b];
      arr[a] = arr[a] ^ arr[b];
    }

    function swapDestruct(arr, a, b) {
      [arr[a], arr[b]] = [arr[b], arr[a]];
    }

    function generateRandomArray(maxSize, maxValue) {
      const size = Math.floor(Math.random() * (maxSize + 1))
      let arr = []
      for (let i = 0; i < size; i++)
        arr.push(Math.floor(Math.random() * (maxValue + 1)))
      return arr;
    }

    function testPerformance(swapFn, name) {
      const testTime = 1000;
      const maxSize = 100000;
      const maxValue = 1000;

      let total = 0;

      for (let t = 0; t < testTime; t++) {
        const arr = generateRandomArray(maxSize, maxValue);
        const start = performance.now();
        for (let i = 1; i < arr.length; i++) {
          swapFn(arr, i - 1, i);
        }
        const end = performance.now();
        total += (end - start);
      }

      console.log(`${name} 平均耗时：${(total / testTime).toFixed(3)} ms`);
    }

    testPerformance(swapXor, "异或交换");
    testPerformance(swapDestruct, "解构交换");
```
